52 enum ExploitabilityRating {
53  EXPLOITABILITY_HIGH, // The crash likely represents
54  // a exploitable memory corruption
55  // vulnerability.
56 
57  EXPLOITABLITY_MEDIUM, // The crash appears to corrupt
58  // memory in a way which may be
59  // exploitable in some situations.
60 
61  EXPLOITABILITY_LOW, // The crash either does not corrupt
62  // memory directly or control over
63  // the effected data is limited. The
64  // issue may still be exploitable
65  // on certain platforms or situations.
66 
67  EXPLOITABILITY_INTERESTING, // The crash does not appear to be
68  // directly exploitable. However it
69  // represents a condition which should
70  // be furthur analyzed.
71 
72  EXPLOITABILITY_NONE, // The crash does not appear to represent
73  // an exploitable condition.
74 
75  EXPLOITABILITY_NOT_ANALYZED, // The crash was not analyzed for
76  // exploitability because the engine
77  // was disabled.
78 
79  EXPLOITABILITY_ERR_NOENGINE, // The supplied minidump's platform does
80  // not have a exploitability engine
81  // associated with it.
82 
83  EXPLOITABILITY_ERR_PROCESSING // An error occured within the
84  // exploitability engine and no rating
85  // was calculated.
86 };

118  // True if the process crashed, false if the dump was produced outside
119  // of an exception handler.
120  bool crashed_;
121 
122  // If the process crashed, the type of crash. OS- and possibly CPU-
123  // specific. For example, "EXCEPTION_ACCESS_VIOLATION" (Windows),
124  // "EXC_BAD_ACCESS / KERN_INVALID_ADDRESS" (Mac OS X), "SIGSEGV"
125  // (other Unix).
126  string crash_reason_;
127 
128  // If the process crashed, and if crash_reason implicates memory,
129  // the memory address that caused the crash. For data access errors,
130  // this will be the data address that caused the fault. For code errors,
131  // this will be the address of the instruction that caused the fault.
132  u_int64_t crash_address_;
133 
134  // If there was an assertion that was hit, a textual representation
135  // of that assertion, possibly including the file and line at which
136  // it occurred.
137  string assertion_;
138 
139  // The index of the thread that requested a dump be written in the
140  // threads vector. If a dump was produced as a result of a crash, this
141  // will point to the thread that crashed. If the dump was produced as
142  // by user code without crashing, and the dump contains extended Breakpad
143  // information, this will point to the thread that requested the dump.
144  // If the dump was not produced as a result of an exception and no
145  // extended Breakpad information is present, this field will be set to -1,
146  // indicating that the dump thread is not available.
147  int requesting_thread_;
148 
149  // Stacks for each thread (except possibly the exception handler
150  // thread) at the time of the crash.
151  vector<CallStack*> threads_;
152  vector<MinidumpMemoryRegion*> thread_memory_regions_;
153 
154  // OS and CPU information.
155  SystemInfo system_info_;
156 
157  // The modules that were loaded into the process represented by the
158  // ProcessState.
159  const CodeModules *modules_;
160 
161  // The exploitability rating as determined by the exploitability
162  // engine. When the exploitability engine is not enabled this
163  // defaults to EXPLOITABILITY_NONE.
164  ExploitabilityRating exploitability_;
165 };
